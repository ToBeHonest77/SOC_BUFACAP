touch filename = create file
git status = give status about local repo
git add . = stage files, not commited yet
git commit -m "first change" = commits files in local repo
git add readme.txt = commit particular file
git rm --cached <filename> = delete particular file

git branch <name> = create branch
git logout <name> = enter branch
git merge <name> = merges all branches
>> change branches and edit files wont change in the other branch

git remote add <origin> <url> = create repo in cloud with url
git push -u <origin> <branch>

mkdir <name> = make directory
git init = initialize local repo
git pull <url> = pull repo from online
